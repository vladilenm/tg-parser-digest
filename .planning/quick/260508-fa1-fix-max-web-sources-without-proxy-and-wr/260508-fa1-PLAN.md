```ts
// quick-260508-fa1: одно-попыточный retry для transient network errors.
// Целимся в undici-классы (UND_ERR_*) и низкоуровневые socket-сбои (ECONNRESET, ETIMEDOUT).
// HTTP non-2xx — НЕ transient: уже throw'ится как `HTTP ${status}` без cause-кода.
function isRetriableFetchError(err: unknown): boolean {
  const code = (err as { cause?: { code?: string } } | null)?.cause?.code;
  if (!code) return false;
  return code.startsWith("UND_ERR_") || code === "ECONNRESET" || code === "ETIMEDOUT";
}
```

  log.info(`[web-scraper] fetch start: ${url} (timeout=${ms}ms)`);
  for (let attempt = 0; attempt < 2; attempt++) {
    const controller = new AbortController();
    const timer = setTimeout(() => controller.abort(), ms);
    const startMs = Date.now();
    try {
      // ...existing redirect loop body unchanged: returns text on success, throws on fail...
      // (paste the entire current `try { ...redirect loop... }` body here, but use `controller`,
      // `timer`, `startMs`, `userAgent`, `ms` from THIS scope — they're already named the same)
    } catch (err) {
      clearTimeout(timer);
      if (attempt === 0 && isRetriableFetchError(err)) {
        const code = (err as { cause?: { code?: string } } | null)?.cause?.code ?? "unknown";
        log.warn(`[web-scraper] fetch retry 1/1 after 5000ms: ${url} (cause: ${code})`);
        await new Promise((r) => setTimeout(r, 5_000));
        continue;
      }

function mskDateYmd(): string {
  return new Intl.DateTimeFormat("en-CA", {

Modify each `log.info/warn/error` to ALSO call `appendToFile` with the same formatted message (preserving the timestamp prefix). Console behavior stays identical (PM2 still gets its lines):

```ts
export const log = {
  info(msg: string, ...ctx: unknown[]): void {
    const line = `[${timestamp()}] [info] ${msg}`;
    console.log(line, ...ctx);
    appendToFile(ctx.length > 0 ? `${line} ${ctx.map((c) => formatCtx(c)).join(" ")}` : line);
  },
  warn(msg: string, ...ctx: unknown[]): void {
    const line = `[${timestamp()}] [warn] ${msg}`;
    console.warn(line, ...ctx);
    appendToFile(ctx.length > 0 ? `${line} ${ctx.map((c) => formatCtx(c)).join(" ")}` : line);
  },
  error(msg: string, ...ctx: unknown[]): void {
    const line = `[${timestamp()}] [error] ${msg}`;
    console.error(line, ...ctx);
    appendToFile(ctx.length > 0 ? `${line} ${ctx.map((c) => formatCtx(c)).join(" ")}` : line);
  },
};

function formatCtx(c: unknown): string {
  if (c instanceof Error) return `${c.name}: ${c.message}${c.stack ? `\n${c.stack}` : ""}`;
  if (typeof c === "string") return c;
  try { return JSON.stringify(c); } catch { return String(c); }
}
```

Reason for `formatCtx`: `console.log(line, ...ctx)` lets Node format errors/objects nicely; `appendFileSync` would otherwise get `[object Object]`. Keep it minimal — string passthrough, Error formatting, JSON.stringify fallback.
