       import type { Workbook } from "exceljs";
       import type { UploadType } from "./types.js";
       const MARKERS: { prefix: string; type: UploadType }[] = [
         { prefix: "цена битум на бирже", type: "birzha_prices" },
         { prefix: "объем битум на бирже", type: "birzha_volumes" },
         { prefix: "битум цены продавцов fca", type: "fca" },
       ];
       export function detectUploadType(wb: Workbook): UploadType | null {
         const ws = wb.worksheets[0];
         if (!ws) return null;
         const a1 = ws.getCell("A1").value;
         const text = typeof a1 === "string" ? a1.trim().toLowerCase() : "";
         if (!text) return null;
         return MARKERS.find(m => text.startsWith(m.prefix))?.type ?? null;
       }

         const dayNr = (target.getUTCDay() + 6) % 7;             // Mon=0..Sun=6
