       void (async () => {

           log.error("bot startBot exited with unexpected error", e);
           const alertId = crypto.randomUUID().slice(0, 8);

               message: e?.message ?? String(err),
               runId: alertId,
               stack: e?.stack,
             });
           } catch (alertErr) {
             log.error("alert send failed", alertErr);
           }
         }
       })();

       const shutdown = async (signal: string): Promise<void> => {
         log.info(`received ${signal}, stopping cron and bot`);
         task.stop();
         // D-06: остановить bot polling и дождаться завершения текущего getUpdates (≤30s timeout polling'а + buffer).
         stopBot();
         const botShutdownDeadline = Date.now() + 35_000; // 30s timeout + 5s buffer
         while (isBotPolling() && Date.now() < botShutdownDeadline) {
           await new Promise((r) => setTimeout(r, 500));
         }
         if (isBotPolling()) {

           await new Promise((r) => setTimeout(r, 500));
         }
         process.exit(0);
       };
